Function: gsl_sf_mathieu_workspace * gsl_sf_mathieu_alloc (size_t n, double qmax)

creates a workspace for the array versions of the Mathieu routines. The arguments n and qmax specify the maximum order and q-value of Mathieu functions which can be computed with this workspace.

static VALUE
sf_mathieu_init(VALUE self, VALUE n, VALUE q)
{
    gsl_sf_mathieu_workspace *w;
    w = gsl_sf_mathieu_alloc(NUM2SIZET(n), NUM2DBL(q));
    if (!w) {
        DATA_PTR(self) = w;
    }
    rb_call_super(0, 0);
    return self;
}
