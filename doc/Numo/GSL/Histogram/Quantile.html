Overview

The functions in this section estimate quantiles dynamically without storing the entire dataset, using the algorithm of Jain and Chlamtec, 1985. Only five points (markers) are stored which represent the minimum and maximum of the data, as well as current estimates of the p/2-, p-, and (1+p)/2-quantiles. Each time a new data point is added, the marker positions and heights are updated.

See “Running Statistics Quantiles”

static VALUE
rquantile_init(VALUE self, VALUE p)
{
    gsl_histogram_quantile_workspace *w, *v;

    Data_Get_Struct(self, gsl_histogram_quantile_workspace, v);
    w = gsl_histogram_quantile_alloc(NUM2DBL(p));
    MEMCPY(v, w, gsl_histogram_quantile_workspace, 1);
    free(w);
    return self;
}

static VALUE
rquantile_add(VALUE self, VALUE data)
{
    gsl_histogram_quantile_workspace *w;
    ndfunc_arg_in_t ain[1] = {{numo_cDFloat,0}};
    ndfunc_t ndf = {iter_rquantile_add, STRIDE_LOOP, 1, 0, ain, 0};

    Data_Get_Struct(self, gsl_histogram_quantile_workspace, w);
    na_ndloop3(&ndf, w, 1, data);
    return self;
}
